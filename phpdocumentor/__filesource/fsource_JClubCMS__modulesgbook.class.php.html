  361.      * in $answer.

  363.      * @param array[reference] $answer Antwort
  364.      * @param array $blacklist Array der Schlüssel, die nicht geprüft werden sollen

  367.     private function _check_form(&$answer, $blacklist = array())
  368.     {
  369.         $gbook_vars = $this->_configvars['Gbook'];
  370.         $error_vars =$this->_configvars['Error'];
  371.  
  372.         /* Formularcheck vorbereiten */
  373.         //$formcheck = new Formularcheck();
  374.  
  375.         /*Formulardaten */
  376.         if (!in_array('title', $blacklist)) {
  377.             /* Titel z.B. bei Kommentar nicht vorhanden */
  378.             $val['title'] = $this->_gpc['POST']['title'];
  379.         }
  380.         $val = array(                            'content' => $this->_gpc['POST']['content'],
  381.         'name' =>  $this->_gpc['POST']['name'], 'email' => $this->_gpc['POST']['email'],
  382.         'hp' => $this->_gpc['POST']['hp']);
  383.         /* Standart-Strings*/
  384.         $std = array('title' => $gbook_vars['entry_title'], 'content' => $gbook_vars['entry_content'],
  385.         'name' => $gbook_vars['entry_name'],'email' => $gbook_vars['entry_email'], 
  386.         'hp' => $gbook_vars['entry_hp']);
  387.         /* Error-Strings */
  388.         $err = array('title' => $error_vars['title_error'], 'content' => $error_vars['content_error'],
  389.         'name' => $error_vars['name_error'],'email' => $error_vars['email_error']);
  390.  
  391.  
  392.         /* Unerwünschte Schlüssel nicht kontrollieren und speichern */
  393.         if (!empty($blacklist) && is_array($blacklist)) {
  394.             foreach ($blacklist as $value) {
  395.                 /* Nur löschen, wenn Variable existiert */
  396.                 if (isset($val[$value])) {
  397.                     unset($val[$value]);
  398.                 }
  399.                 if (isset($std[$value])) {
  400.                     unset($std[$value]);
  401.                 }
  402.                 if (isset($err[$value])) {
  403.                     unset($err[$value]);
  404.                 }
  405.             }
  406.         }
  407.  
  408.         $rtn_arr = $this->_msbox->formCheck($val, $std);
  409.  
  410.         
  411.         /* Fehlerarray durchgehen */
  412.         foreach ($rtn_arr as $key => $value) {
  413.             if ($value == MSGBOX_FORMCHECK_NONE) {
  414.                 $answer[] = $err[$key];
  415.             }
  416.  
  417.             if ($value == MSGBOX_FORMCHECK_INVALID && $key = 'email') {
  418.                 $answer[] = $error_vars['email_checkfailed'];
  419.             } elseif ($key == 'hp' ) {
  420.                 /* Wenn die hp falsch ist, gibt es kein Fehler */
  421.                     $val[$key] = $rtn_arr[$key];    
  422.             }
  423.         }
  424.  
  425.         /* Captcha-Image prüfen */
  426.         if(!$this->_captcha->verify($this->_gpc['POST']['captcha_word'])) {
  427.             $answer[] = $error_vars['captcha_error']."<br />";
  428.         }
  429.         
  430.         if (empty($answer)) {
  431.             if (!in_array('title', $blacklist)) {
  432.                 $answer['title'] = $this->_gpc['POST']['title'];
  433.             }
  434.  
  435.             /*Wenn keine Fehler aufgetaucht sind, werden die Einträge zurückgegeben*/
  436.             $answer += array('content' => $val['content'], 'name' => $val['name'],
  437.             'time' => 'gbook_time', 'email' => $val['email'], 'hp' => $val['hp']);
  438.  
  439.             return true;
  440.         } else {
  441.             return false;
  442.         }
  443.     }
