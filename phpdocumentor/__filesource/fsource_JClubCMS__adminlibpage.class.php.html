  465.                 }

  467.                 $level--;
  468.  
  469.             } elseif (count($topid_array) > 0 && $value['menu_ID'] == $topid_array[0]) {
  470.                 $this->_build_subnav_array($table_name, $topid_array, $subnav_array);

  474.         }
